```html
<!DOCTYPE html>
<!-- 
  文档头（类型）声明
  作用：告知当前网页内容，采用何种解析方式
  注意：1.文档头不是标签
       2.html代表采用html5的标准
       3.文档头放在最前面
-->
<html>
  <!-- html根元素，所有html代码等内容都需要放在根元素中 -->
  <head>
    <!-- head:头部，辅助‘身体部分’，进行展示 -->
    <title>我是标题</title>
    <!-- title:设置页面标题 -->
    <meta charset="utf-8" />
    <!-- meta:单标签，定义页面的元数据
 属性：charset：针对搜索引擎和解析方式使用的属性
-->
  </head>
  <body>
    <!--body:身体，绝大部分内容都在此展示 -->
  </body>
</html>
```
